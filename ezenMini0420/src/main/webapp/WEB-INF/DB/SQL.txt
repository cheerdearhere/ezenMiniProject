**유저 DB
CREATE TABLE EZENUSER 
(
  PID VARCHAR2(30 BYTE) NOT NULL 
, PPW VARCHAR2(120 BYTE) NOT NULL 
, PADDRESS VARCHAR2(150 BYTE) 
, COLUMN2 VARCHAR2(20) 
, PPROFILE VARCHAR2(300) 
, PDATE DATE DEFAULT(sysdate)
, AUTH VARCHAR2(20) DEFAULT('ROLE_USER')
, CONSTRAINT EZENUSER_PK PRIMARY KEY 
  (PID);

**recipe DB
CREATE SEQUENCE RECIPE_SEQ NOCACHE;
  
CREATE TABLE RECIPE 
(
  RID NUMBER NOT NULL 
, RCLASS VARCHAR2(30) NOT NULL 
, RTRNAME VARCHAR2(30) NOT NULL 
, RTITLE VARCHAR2(30) NOT NULL 
, RPHOTO VARCHAR2(100) NOT NULL 
, RCONTENT VARCHAR2(300) NOT NULL 
, CONSTRAINT RECIPE_PK PRIMARY KEY 
  (
    RID 
  )
  ENABLE 
);

**시퀀스용
CREATE SEQUENCE FullCalendar_SEQ NOCACHE;

**FullCalendar DB

CREATE TABLE FULLCALENDAR 
(
  CNO NUMBER(4) NOT NULL 
, CID VARCHAR2(30) NOT NULL 
, CTITLE VARCHAR2(100) NOT NULL 
, CSTART VARCHAR2(100) NOT NULL 
, CEND VARCHAR2(100) NOT NULL 
, TSTART DATE 
, TEND DATE 
, CALLDAY VARCHAR2(100) 
, CONSTRAINT FULLCALENDAR_PK PRIMARY KEY 
  (
    CNO 
  )
  ENABLE 
);

CREATE TABLE RECIPEORDER
